wallet-panel.js — WalletPanel.refresh(uid)
function refresh(uid) { if (uid) _uid = uid; /* Always update the navbar badge immediately — even if the wallet panel tab was never opened and _core is not yet initialized. */ if (_uid && typeof AppService !== 'undefined' && AppService.getWallet) { AppService.getWallet(_uid, function(err, wallet) { if (!err && wallet) _onBalanceUpdate(wallet); }); } /* If panel is already mounted and core exists, reload full panel data in-place. If not yet mounted (wallet tab not visited), nothing more to update — mount() will load fresh data when the tab is opened. */ if (_core) { _core.loadData(); } }

function _onBalanceUpdate(wallet) { _wallet = wallet; var badge = document.getElementById('navbar-wallet-amount'); if (badge && wallet) { var _nbCur = (typeof CurrencyService !== 'undefined' && typeof ProfileStore !== 'undefined') ? CurrencyService.getUserCurrency(_uid) : 'EUR'; var _nbConv = (_nbCur !== 'EUR' && typeof CurrencyService !== 'undefined') ? CurrencyService.convertSync(wallet.balance, 'EUR', _nbCur) : null; badge.textContent = _nbConv !== null ? CurrencyService.format(_nbConv, _nbCur) : WalletCore.formatAmount(wallet.balance) + ' €'; } }

_core-Zustand — wann ist das Wallet-Panel gemountet?
Szenario_coreNavbar-BadgeWallet-Panel
Wallet-Tab wurde noch nicht besucht null ✅ Wird via AppService.getWallet() aktualisiert Wird beim nächsten Tab-Besuch frisch geladen
Wallet-Tab wurde mindestens einmal besucht WalletCore-Instanz ✅ Wird via _core.loadData() → _onBalanceUpdate() aktualisiert ✅ Vollständig neu geladen (Transaktionsliste + Balance)
Doppelter _onBalanceUpdate-Aufruf: Wenn _core vorhanden ist, wird _onBalanceUpdate() zweimal aufgerufen — einmal direkt via AppService.getWallet() und einmal durch _core.loadData(). Das ist harmlos: beide schreiben denselben Wert in das Badge-Element. Der direkte Aufruf ist schneller (kein TX-Ladevorgang), der Core-Aufruf aktualisiert zusätzlich die Transaktionsliste.
