Callback-Barrier für parallele Async-Ops — student.js + teacher.js R1 · Flackernde UI + Race Condition bei Mehrfach-Buchung / Mehrfach-Storno
~25 Zeilen je Datei
Prinzip: Alle Async-Calls feuern weiterhin parallel (kein sequenzielles Warten). Ein Counter pending zählt herunter. Erst wenn alle Callbacks zurück sind, wird einmal re-gerendert. Errors werden gesammelt und nach dem Barrier als Toast ausgegeben.

Muster (gilt für beide Dateien)

function commitDayChanges() {
  var ids    = Object.keys(pendingDayChanges);
  var saved  = {};
  var pending = ids.length;
  var errors  = [];

  function onAllDone() {
    if (errors.length) Toast.error(errors[0]);
    pendingDayChanges = {};
    updateDaySaveBtn();
    renderStats();
    _updateMyBookingsBadges();
    renderMyBookingsList();
    renderCalendar();
    renderDaySlots();
    _refreshNavbarWallet(currentUser.uid);
    if (typeof WalletPanel !== 'undefined') WalletPanel.refresh(currentUser.uid);
    if (!errors.length) Toast.success('Gespeichert.');
  }

  if (!pending) { onAllDone(); return; }  /* nothing to do */

  for (var i = 0; i < ids.length; i++) {
    var p = pendingDayChanges[ids[i]];
    saved[ids[i]] = p;
    if (p.action === 'book') {
      (function(sid, stuId, tid) {
        AppService.bookSlotWithEscrow(sid, stuId, tid, function(e) {
          if (e) errors.push(e.message || e);
          /* individual re-renders removed — onAllDone handles it */
          if (--pending === 0) onAllDone();
        });
      })(ids[i], p.newStudentId || currentUser.uid, p.originalSlot.teacherId);
    } else {
      (function(sid) {
        var slot = AppService.getAllSlotsSync().filter(function(s){ return s.slotId === sid; })[0];
        if (!slot || slot.status !== 'booked') {
          if (--pending === 0) onAllDone();
          return;
        }
        AppService.cancelSlotWithPolicy(sid, 'student', function(e) {
          if (e) errors.push(e.message || e);
          if (--pending === 0) onAllDone();
        });
      })(ids[i]);
    }
  }
  /* Email notifications fire independently — no barrier needed */
  /* ... email loop unchanged ... */
  /* Remove the old synchronous render block that was here */
}

Strategie

/* 1. renderAllBookingsList(opts) — neues optionales Argument */
function renderAllBookingsList(opts) {
  opts = opts || {};
  var dateFilter = opts.dateFilter || null;  /* null = alle Tage */

  var allSlots = AppService.getSlotsByTeacherSync(currentUser.uid)
    .filter(function(s) { return s.status === 'booked' && s.studentId; });

  if (dateFilter) allSlots = allSlots.filter(function(s) { return s.date === dateFilter; });

  // ... rest of pipeline unchanged ...
}

/* 2. renderBookingsPanel wird zum einzeiligen Delegator */
function renderBookingsPanel(panel, slots, dateStr) {
  panel.innerHTML = '';
  renderAllBookingsList({ dateFilter: dateStr, container: panel });
}
