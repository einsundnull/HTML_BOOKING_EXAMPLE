student.js — buildTeacherCard() · Zeile ~946 (Punkt 1 + Badge-Logik)
/* Vorher */ var price = AppService.getTeacherPriceSync(teacher.uid) || null; /* Nachher */ var effectivePrice = currentUser ? AppService.getStudentPriceForTeacherSync(currentUser.uid, teacher.uid) : AppService.getTeacherPriceSync(teacher.uid); var price = effectivePrice || null; var defaultPrice = AppService.getTeacherPriceSync(teacher.uid); var hasIndividual = currentUser && price !== null && price !== defaultPrice; /* Im Preis-Element: */ if (hasIndividual) { priceEl.className = 'teacher-card-price-individual'; priceEl.innerHTML = priceText + ' <span class="price-individual-badge">Ihr Preis</span>'; } else { priceEl.className = 'teacher-card-price'; priceEl.textContent = priceText; }
